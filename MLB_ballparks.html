scrape_coordinate_wiki <- function(url) {
  webc <- rvest::read_html(url)
  webtd <- webc %>% 
  rvest::html_nodes("td") %>%
  rvest::html_text()
  coordinatetd <- webtd[grepl("Coordinates", webtd)]
  coord <- gsub("^.*\\Coor","Coor",coordinatetd)
  coor <- gsub(".*\\¬∞WÔªø / ","",coord)
  coordinate <- data.frame(
    url = url,
    coor = coor
  )
  return(coordinate)
}
